 22 #define get_seg_byte(seg,addr) ({ \
 23 register char __res; \
 24 __asm__("push %%fs;mov %%ax,%%fs;movb %%fs:%2,%%al;pop %%fs" \
 25     :"=a" (__res):"0" (seg),"m" (*(addr))); \
 26 __res;})
 27 
 28 #define get_seg_long(seg,addr) ({ \
 29 register unsigned long __res; \
 30 __asm__("push %%fs;mov %%ax,%%fs;movl %%fs:%2,%%eax;pop %%fs" \
 31     :"=a" (__res):"0" (seg),"m" (*(addr))); \
 32 __res;})
 33 
 34 #define _fs() ({ \
 35 register unsigned short __res; \
 36 __asm__("mov %%fs,%%ax":"=a" (__res):); \
 37 __res;})

 62 static void die(char * str,long esp_ptr,long nr)
 63 {
 64     long * esp = (long *) esp_ptr;
 65     int i;
 66 
 67     printk("%s: %04x\n\r",str,nr&0xffff);
 68     printk("EIP:\t%04x:%p\nEFLAGS:\t%p\nESP:\t%04x:%p\n",
 69         esp[1],esp[0],esp[2],esp[4],esp[3]);
 70     printk("fs: %04x\n",_fs());
 71     printk("base: %p, limit: %p\n",get_base(current->ldt[1]),get_limit(0x17));
 72     if (esp[4] == 0x17) {
 73         printk("Stack: ");
 74         for (i=0;i<4;i++)
 75             printk("%p ",get_seg_long(0x17,i+(long *)esp[3]));
 76         printk("\n");
 77     }
 78     str(i);
 79     printk("Pid: %d, process nr: %d\n\r",current->pid,0xffff & i);
 80     for(i=0;i<10;i++)
 81         printk("%02x ",0xff & get_seg_byte(esp[1],(i+(char *)esp[0])));
 82     printk("\n\r");
 83     do_exit(11);        /* play segment exception */
 84 }

106 void do_int3(long * esp, long error_code,
107         long fs,long es,long ds,
108         long ebp,long esi,long edi,
109         long edx,long ecx,long ebx,long eax)
110 {
111     int tr;
112 
113     __asm__("str %%ax":"=a" (tr):"0" (0));
114     printk("eax\t\tebx\t\tecx\t\tedx\n\r%8x\t%8x\t%8x\t%8x\n\r",
115         eax,ebx,ecx,edx);
116     printk("esi\t\tedi\t\tebp\t\tesp\n\r%8x\t%8x\t%8x\t%8x\n\r",
117         esi,edi,ebp,(long) esp);
118     printk("\n\rds\tes\tfs\ttr\n\r%4x\t%4x\t%4x\t%4x\n\r",
119         ds,es,fs,tr);
120     printk("EIP: %8x   CS: %4x  EFLAGS: %8x\n\r",esp[0],esp[1],esp[2]);
121 }

185 /* 设置中断描述符表 */
186 void trap_init(void)
187 {
188     int i;
189 
190     set_trap_gate(0,&divide_error); /* 除 0 中断 */
191     set_trap_gate(1,&debug); /* 调试 */
192     set_trap_gate(2,&nmi); /* 不可屏蔽中断 */
193     set_system_gate(3,&int3);   /* 调试, int3-5 可以被所有程序执行 */
194     set_system_gate(4,&overflow); /* into 溢出 */
195     set_system_gate(5,&bounds); /* bound 检查错误 */
196     set_trap_gate(6,&invalid_op); /* 无效指令 */
197     set_trap_gate(7,&device_not_available); /* 找不到设备 */
198     set_trap_gate(8,&double_fault); /* 双重错误 */
199     set_trap_gate(9,&coprocessor_segment_overrun); /* 协处理器段超出错误 */
200     set_trap_gate(10,&invalid_TSS); /* 非法 TSS */
201     set_trap_gate(11,&segment_not_present); /* 段不存在 */
202     set_trap_gate(12,&stack_segment); /* 堆栈段错误 */
203     set_trap_gate(13,&general_protection); /* 通用保护 */
204     set_trap_gate(14,&page_fault); /* 页面异常 */
205     set_trap_gate(15,&reserved); /* 保留 */
206     set_trap_gate(16,&coprocessor_error); /* 协处理器错误 */
207     set_trap_gate(17,&alignment_check); /* 对齐错误 */
208 
209     /* int17 ~ int47 的陷阱门先均设置为 reserved
210      * 以后各硬件初始化时会重新设置自己的陷阱门 */
211     for (i=18;i<48;i++)
212         set_trap_gate(i,&reserved);
213 
214     /* 设置协处理器中断 0x2d(即45)陷阱门描述符, 并允许其产生中断请求, 设置并行口中断描述符 */
215     set_trap_gate(45,&irq13);
216 
217     outb_p(inb_p(0x21)&0xfb,0x21); /* 允许 8259A 主芯片的 IRQ2 中断请求 */
218     outb(inb_p(0xA1)&0xdf,0xA1);  /* 允许 8259A 从芯片的 IRQ13 中断请求 */
219 
220     set_trap_gate(39,&parallel_interrupt); /* 设置并行口 1 的中断 0x27 陷阱门描述符 */
221 }
